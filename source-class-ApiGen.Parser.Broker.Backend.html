 82:     /**
 83:      * Prepares and returns used class lists.
 84:      *
 85:      * @return ReflectionClass[]
 86:      */
 87:     protected function parseClassLists()
 88:     {
 89:         $this->declared = array_flip(array_merge(get_declared_classes(), get_declared_interfaces()));
 90: 
 91:         foreach ($this->getNamespaces() as $namespace) {
 92:             foreach ($namespace->getClasses() as $name => $ref) {
 93:                 $class = $this->reflectionFactory->createFromReflection($ref);
 94: 
 95:                 $this->allClasses[self::TOKENIZED_CLASSES][$name] = $class;
 96:                 if ( ! $class->isDocumented()) {
 97:                     continue;
 98:                 }
 99: 
100:                 $this->loadParentClassesAndInterfacesFromClassReflection($ref);
101:             }
102:         }
103: 
104:         /** @var ReflectionClass $class */
105:         foreach ($this->allClasses[self::TOKENIZED_CLASSES] as $class) {
106:             if ( ! $class->isDocumented()) {
107:                 continue;
108:             }
109: 
110:             foreach ($class->getOwnMethods() as $method) {
111:                 $this->processFunction($method);
112:             }
113: 
114:             foreach ($class->getOwnProperties() as $property) {
115:                 $this->loadAnnotationFromReflection($class, $property->getAnnotations(), 'var');
116:             }
117:         }
118: 
119:         foreach ($this->getFunctions() as $function) {
120:             $this->processFunction($function);
121:         }
122: 
123:         array_walk_recursive($this->allClasses, function (&$reflection) {
124:             if ( ! $reflection instanceof ReflectionClass) {
125:                 $reflection = $this->reflectionFactory->createFromReflection($reflection);
126:             }
127:         });
128: 
129:         return $this->allClasses;
130:     }

133:     /**
134:      * Processes a function/method and adds classes from annotations to the overall class array.

137:      */

139:     {
140:         $annotations = $reflection->getAnnotations();

143:         }

156:     private function addClass($name)
157:     {
158:         $name = ltrim($name, '\\');
159: 
160:         if ( ! isset($this->declared[$name]) || $this->isClassLoaded($name)) {
161:             return FALSE;
162:         }
163: 
164:         $parameterClass = $this->getBroker()->getClass($name);
165: 
166:         if ($parameterClass->isInternal()) {
167:             $this->allClasses[self::INTERNAL_CLASSES][$name] = $parameterClass;
168:             foreach (array_merge($parameterClass->getInterfaces(), $parameterClass->getParentClasses()) as $parentClass) {
169:                 if ( ! isset($this->allClasses[self::INTERNAL_CLASSES][$parentName = $parentClass->getName()])) {
170:                     $this->allClasses[self::INTERNAL_CLASSES][$parentName] = $parentClass;
171:                 }
172:             }
173: 
174:         } elseif ( ! $parameterClass->isTokenized()) {
175:             if ( ! isset($this->allClasses[self::NONEXISTENT_CLASSES][$name])) {
176:                 $this->allClasses[self::NONEXISTENT_CLASSES][$name] = $parameterClass;
177:             }
178:         }
179:     }

186:     {
187:         foreach (array_merge($ref->getParentClasses(), $ref->getInterfaces()) as $parentName => $parentReflection) {

189:             if ($parentReflection->isInternal()) {
190:                 if ( ! isset($this->allClasses[self::INTERNAL_CLASSES][$parentName])) {
191:                     $this->allClasses[self::INTERNAL_CLASSES][$parentName] = $parentReflection;
192:                 }
193: 
194:             } elseif ( ! $parentReflection->isTokenized()) {
195:                 if ( ! isset($this->allClasses[self::NONEXISTENT_CLASSES][$parentName])) {
196:                     $this->allClasses[self::NONEXISTENT_CLASSES][$parentName] = $parentReflection;
197:                 }
198:             }
199:         }
200:     }

215:     /**
216:      * @param ReflectionClass|ReflectionMethod $reflection
217:      * @param array $annotations
218:      * @param string $name
219:      */
220:     private function loadAnnotationFromReflection($reflection, array $annotations, $name)
221:     {
222:         if ( ! isset($annotations[$name])) {
223:             return;
224:         }
225: 
226:         foreach ($annotations[$name] as $doc) {
227:             foreach (explode('|', preg_replace('~\\s.*~', '', $doc)) as $name) {
228:                 if ($name = rtrim($name, '[]')) {
229:                     $name = $this->getClassFqn($name, $reflection);

231:                 }
232:             }
233:         }
234:     }
