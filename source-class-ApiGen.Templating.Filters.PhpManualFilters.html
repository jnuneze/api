 27: 
 28:     /**
 29:      * @var array [ className => callback ]
 30:      */
 31:     private $assignments = [];
 32: 
 33: 
 34:     public function __construct()
 35:     {
 36:         $this->prepareAssignments();
 37:     }

 40:     /**
 41:      * @param ReflectionElement|ReflectionExtension|ReflectionMethod $element
 42:      * @return string
 43:      */
 44:     public function manualUrl($element)
 45:     {
 46:         if ($element instanceof ReflectionExtension) {
 47:             return $this->createExtensionUrl($element);
 48:         }
 49: 
 50:         $class = $this->detectClass($element);
 51:         if ($class && $this->isReservedClass($class)) {
 52:             return self::PHP_MANUAL_URL . '/reserved.classes.php';
 53:         }
 54: 
 55:         $className = get_class($element);
 56:         if (isset($this->assignments[$className])) {
 57:             return $this->assignments[$className]($element, $class);
 58:         }
 59:         return '';
 60:     }
 61: 
 62: 
 63:     /**
 64:      * @return string
 65:      */
 66:     private function createExtensionUrl(ReflectionExtension $reflectionExtension)
 67:     {
 68:         $extensionName = strtolower($reflectionExtension->getName());
 69:         if ($extensionName === 'core') {
 70:             return self::PHP_MANUAL_URL;
 71:         }
 72: 
 73:         if ($extensionName === 'date') {
 74:             $extensionName = 'datetime';
 75:         }
 76: 
 77:         return self::PHP_MANUAL_URL . '/book.' . $extensionName . '.php';
 78:     }
 79: 
 80: 
 81:     /**
 82:      * @return array
 83:      */
 84:     private function prepareAssignments()
 85:     {
 86:         $this->assignments['ApiGen\Reflection\ReflectionClass'] = function ($element, $class) {
 87:             return $this->createClassUrl($class);
 88:         };
 89:         $this->assignments['ApiGen\Reflection\ReflectionMethod'] = function ($element, $class) {
 90:             return $this->createMethodUrl($element, $class);
 91:         };
 92:         $this->assignments['ApiGen\Reflection\ReflectionFunction'] = function ($element, $class) {
 93:             return $this->createFunctionUrl($element);
 94:         };
 95:         $this->assignments['ApiGen\Reflection\ReflectionProperty'] = function ($element, $class) {
 96:             return $this->createPropertyUrl($element, $class);
 97:         };
 98:         $this->assignments['ApiGen\Reflection\ReflectionConstant'] = function ($element, $class) {
 99:             return $this->createConstantUrl($element, $class);
100:         };
101:     }
